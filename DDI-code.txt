import csv


def load_data(path):
    """
    Lädt DDI-Daten aus einer CSV-Datei und speichert sie
    als Dictionary für schnelle Abfragen.
    """
    interactions = {}

    with open(path, newline='') as f:
        reader = csv.DictReader(f)

        for row in reader:
            d1 = row['drug1'].lower()
            d2 = row['drug2'].lower()

            # Speichere beide Kombinationen (A+B und B+A)
            interactions[(d1, d2)] = row
            interactions[(d2, d1)] = row

    return interactions


def check_interactions(drugs, interactions):
    """
    Prüft alle möglichen Paare von Medikamenten
    auf bekannte Interaktionen.
    """
    results = []

    for i in range(len(drugs)):
        for j in range(i + 1, len(drugs)):
            pair = (drugs[i], drugs[j])

            if pair in interactions:
                results.append(interactions[pair])

    return results


def main():
    # Daten laden
    data = load_data('data/ddi_data.csv')

    # Nutzereingabe
    user_input = input("Enter drugs (comma separated): ")
    drugs = [d.strip().lower() for d in user_input.split(',') if d.strip()]

    # Interaktionen prüfen
    results = check_interactions(drugs, data)

    # Ergebnisse ausgeben
    if results:
        print("\nInteractions found:\n")

        for r in results:
            print(f"{r['drug1']} + {r['drug2']}")
            print(f"Severity: {r['severity']}")
            print(f"Recommendation: {r['recommendation']}\n")
    else:
        print("No interactions found.")


if __name__ == '__main__':
    main()
